def start_openclaw():
    try:
        # Check if OpenClaw is running
        result = subprocess.run(

        )
        if "openclaw.exe" in result.stdout.lower():
            logger.info("OpenClaw already running.")
            return None
        else:

            return subprocess.Popen([OPENCLAW_CMD, "start"], cwd=BASE_DIR)
    except Exception as e:

        return subprocess.Popen([sys.executable, BOT_SCRIPT], cwd=BASE_DIR)
    except Exception as e:
        logger.error(f"Failed to start bot.py: {e}")
        return None

# --- Launcher Loop ---
def main():
    openclaw_proc = start_openclaw()
    bot_proc = start_bot()

    try:
        while True:
            time.sleep(5)
            # Restart OpenClaw if it crashed
            if openclaw_proc and openclaw_proc.poll() is not None:
                logger.warning("OpenClaw crashed, restarting...")
                openclaw_proc = start_openclaw()

            # Restart bot if it crashed
            if bot_proc and bot_proc.poll() is not None:
                logger.warning("bot.py crashed, restarting...")
                bot_proc = start_bot()
    except KeyboardInterrupt:
        logger.info("Shutting down launcher...")
        if openclaw_proc:
            openclaw_proc.terminate()
        if bot_proc:
            bot_proc.terminate()

if __name__ == "__main__":
    main()

        logging.StreamHandler(sys.stdout)
    ]

# --- Functions ---
def start_openclaw():
    try:
        logger.info("Checking if OpenClaw is running...")
        result = subprocess.run(["tasklist"], capture_output=True, text=True)
        if "openclaw.exe" in result.stdout.lower():
            logger.info("OpenClaw is already running.")
            return None
        logger.info(f"Starting OpenClaw using {OPENCLAW_CMD} ...")
        proc = subprocess.Popen([OPENCLAW_CMD, "start"], cwd=OPENCLAW_CWD)
        return proc
    except Exception as e:
        logger.error(f"Failed to start OpenClaw: {e}")
        return None

def start_bot():
    try:
        logger.info("Starting bot.py ...")
        proc = subprocess.Popen([sys.executable, BOT_SCRIPT], cwd=BASE_DIR)
        return proc
    except Exception as e:
        logger.error(f"Failed to start bot.py: {e}")
        return None

# --- Main Launcher Loop ---
def main():
    logger.info(f"KonstanceAI launcher started (ROOT={BASE_DIR})")
    openclaw_proc = start_openclaw()
    bot_proc = start_bot()

    logger.info("Launcher is monitoring bot.py and OpenClaw processes...")
    try:
        while True:
            time.sleep(5)
            if openclaw_proc and openclaw_proc.poll() is not None:
                logger.warning("OpenClaw crashed, restarting...")
                openclaw_proc = start_openclaw()
            if bot_proc and bot_proc.poll() is not None:
                logger.warning("bot.py crashed, restarting...")
                bot_proc = start_bot()
    except KeyboardInterrupt:
        logger.info("Shutting down launcher and subprocesses...")
        if openclaw_proc: openclaw_proc.terminate()
        if bot_proc: bot_proc.terminate()

if __name__ == "__main__":
    main()

import subprocess

import os
import datetime
import shutil
from pathlib import Path
from telegram import Update
from telegram.ext import Updater, CommandHandler, CallbackContext, MessageHandler, Filters

# ----------------------------
# Configuration
# ----------------------------
KONSTANCE_ROOT = Path(r"C:\Users\Thinkpad\Desktop\KonstanceAI")
BACKUP_ROOT = KONSTANCE_ROOT / "backups"
MASTER_PROMPT_PATH = KONSTANCE_ROOT / "master_prompt.txt"  # store your master prompt here
OLLAMA_MODEL = "qwen2.5:14b"

# Telegram bot token
TELEGRAM_TOKEN = os.environ.get("TELEGRAM_BOT_TOKEN") or open(KONSTANCE_ROOT / "telegram_token.txt").read().strip()

# ----------------------------
# Helper Functions
# ----------------------------
def ensure_aider_installed():
    try:
        subprocess.run(["aider", "--version"], check=True, stdout=subprocess.PIPE, stderr=subprocess.PIPE)
        print("[INFO] Aider is already installed.")
    except (subprocess.CalledProcessError, FileNotFoundError):
        print("[INFO] Aider not found. Installing via pip...")
        subprocess.run([sys.executable, "-m", "pip", "install", "aider-install"], check=True)
        print("[INFO] Aider installed successfully.")

def backup_konstance_folder():
    timestamp = datetime.datetime.now().strftime("%Y-%m-%d_%H%M%S")
    backup_path = BACKUP_ROOT / timestamp
    backup_path.mkdir(parents=True, exist_ok=True)
    print(f"[INFO] Creating backup at {backup_path}")
    for item in KONSTANCE_ROOT.iterdir():
        if item.name != "backups":  # avoid recursive copy
            dest = backup_path / item.name
            if item.is_dir():
                shutil.copytree(item, dest)
            else:
                shutil.copy2(item, dest)
    return backup_path

def run_aider_with_prompt(user_prompt: str):
    print("[INFO] Running Aider with master prompt...")
    # Save user prompt to temporary file
    temp_prompt = KONSTANCE_ROOT / "aider_prompt.txt"
    temp_prompt.write_text(user_prompt, encoding="utf-8")
    result = subprocess.run([
        "aider", "run",
        "--folder", str(KONSTANCE_ROOT),
        "--model", f"ollama/{OLLAMA_MODEL}",
        "--prompt", str(temp_prompt)
    ], capture_output=True, text=True)
    if result.returncode != 0:
        print("[ERROR] Aider failed:", result.stderr)
        return result.stderr
    return result.stdout

def update_recap(summary: str):
    recap_file = KONSTANCE_ROOT / "recap.txt"
    recap_file.write_text(summary, encoding="utf-8")
    print("[INFO] recap.txt updated.")

# ----------------------------
# Telegram Command Handlers
# ----------------------------
def handle_message(update: Update, context: CallbackContext):
    user_prompt = update.message.text
    update.message.reply_text("[INFO] Received task. Preparing Konstance sub-agent...")

    # Step 1: Ensure Aider is installed
    ensure_aider_installed()

    # Step 2: Backup current folder
    backup_path = backup_konstance_folder()

    # Step 3: Run Aider with the prompt
    output = run_aider_with_prompt(user_prompt)

    # Step 4: Update recap.txt
    recap_summary = f"Task: {user_prompt}\nBackup: {backup_path}\nAider Output:\n{output}\n"
    update_recap(recap_summary)

    # Step 5: Respond to Telegram
    update.message.reply_text("[INFO] Sub-agent completed the task.\nOutput:\n" + output[:2000])  # Telegram limit

# ----------------------------
# Main
# ----------------------------
def main():
    updater = Updater(token=TELEGRAM_TOKEN, use_context=True)
    dp = updater.dispatcher
    dp.add_handler(MessageHandler(Filters.text & ~Filters.command, handle_message))
    updater.start_polling()
    print("[INFO] Konstance Telegram bot running...")
    updater.idle()

if __name__ == "__main__":
    main()

Script:

Write-Host "===== KONSTANCE SYSTEM REPAIR + LAUNCH =====" -ForegroundColor Cyan

$root = "C:\Users\Thinkpad\Desktop\KonstanceAI"
$ollamaModel = "qwen2.5:7b"

# Ensure PATH contains aider
$env:PATH = "C:\Users\Thinkpad\.local\bin;$env:PATH"

# Move to project directory
Set-Location $root
Write-Host "Working directory set to $root" -ForegroundColor Green

# Create required folders
$folders = @("logs","data","backups")
foreach ($f in $folders) {
    if (!(Test-Path $f)) {
        New-Item -ItemType Directory -Path $f | Out-Null
        Write-Host "Created folder: $f"
    }
}

# Check Python
Write-Host "Checking Python..."
python --version

# Install requirements if file exists
if (Test-Path "requirements.txt") {
    Write-Host "Installing requirements..."
    pip install -r requirements.txt
}

# Ensure Aider installed
Write-Host "Checking Aider..."
try {
    aider --version | Out-Null
    Write-Host "Aider already installed"
}
catch {
    Write-Host "Installing Aider..."
    python -m pip install aider-install
    aider-install
}

# Start Ollama if not running
Write-Host "Checking Ollama service..."

$ollama = Get-Process ollama -ErrorAction SilentlyContinue

if (!$ollama) {
    Write-Host "Starting Ollama server..."
    Start-Process "ollama" "serve"
    Start-Sleep 5
}

# Install model if missing
Write-Host "Checking Ollama model..."

$modelList = ollama list

if ($modelList -notmatch $ollamaModel) {
    Write-Host "Downloading model $ollamaModel ..."
    ollama pull $ollamaModel
}

# Create llm config if missing
$llmConfig = "$root\data\llm.json"

if (!(Test-Path $llmConfig)) {
    Write-Host "Creating llm.json config..."
    $json = @"
{
  "primary": "ollama/$ollamaModel",
  "fallback": "ollama/$ollamaModel"
}
"@
    $json | Out-File $llmConfig
}

# Backup project
$date = Get-Date -Format "yyyy-MM-dd_HHmm"
$backup = "$root\backups\$date"

Write-Host "Creating backup..."
Copy-Item $root $backup -Recurse -Exclude backups

# Launch Konstance
Write-Host "Launching Konstance AI..."

Start-Process python "launcher.py"

Write-Host ""
Write-Host "========================================="
Write-Host "KONSTANCE IS STARTING" -ForegroundColor Green
Write-Host "You should now be able to message it on Telegram."
Write-Host "========================================="
